conf t
hostname EDGE_1

! Giao diện kết nối ra INTERNET (outside)
interface GigabitEthernet0/0
ip address 100.0.0.1 255.255.255.0
ip nat outside
no shutdown
exit

! Giao diện kết nối vào ASA (inside)
interface GigabitEthernet0/1
ip address 192.168.1.1 255.255.255.252
ip nat inside
no shutdown
exit

interface GigabitEthernet0/2
ip address 192.168.1.29 255.255.255.252
ip nat inside
no shutdown
exit

! Cấu hình OSPF để tham gia định tuyến động
router ospf 1
router-id 1.1.1.5
network 192.168.1.0 0.0.0.3 area 0
network 192.168.1.28 0.0.0.3 area 0
default-information originate
exit

! Access-list NAT: loại trừ subnet kết nối ASA và cho phép NAT VLAN
access-list 1 deny 192.168.1.0 0.0.0.3    
access-list 1 deny 192.168.1.8 0.0.0.3
access-list 1 deny 192.168.1.12 0.0.0.3
access-list 1 deny 192.168.1.24 0.0.0.3
access-list 1 permit 192.168.0.0 0.0.255.255

! NAT Overload (PAT)
ip nat inside source list 1 interface Gi0/0 overload

! Default route ra Internet
ip route 0.0.0.0 0.0.0.0 100.0.0.2

end
write memory
